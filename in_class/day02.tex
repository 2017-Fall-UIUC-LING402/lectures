\frame {
\frametitle{Practice basic git workflow (\homedir20 minutes)}

	\begin{itemize}
		\item Fork https://github.com/mjdenkowski/lunchpy on Github
		\item Clone your fork
		\item Make some changes
		\item git add
		\item git commit
		\item git push
		\item Repeat a few times with more changes
	\end{itemize}

}



\frame {
\frametitle{Practice text editing (\homedir15 minutes)}

	\begin{itemize}
		\item Create a hidden file in your home directory called \homedir/.lunchrc
		\item Edit the file (using vim)
		\item Add the names of your favorite places to eat lunch (one per line)
		\item Save the file
		\item Exit the editor
	\end{itemize}

}


\frame {
\frametitle{Run a command (\homedir15 minutes)}

	\begin{itemize}
		\item Run lunchpy
		\item Create a hidden directory in your home directory called \homedir/.local/bin
		\item Edit the .bashrc file in your home directory
		\item Add \homedir/.local/bin to your PATH
		\item Exit the editor
		\item Exit the server
		\item Log in again
		\item Run lunchpy
	\end{itemize}

}


\frame {
\frametitle{Advanced option}

	If you get through the previous work quickly, practice git merging:

	\begin{itemize}
		\item Clone someone else's fork of lunchpy
		\item Have them give you write permission on Github
		\item Make some changes
		\item Have them make changes
		\item One of you push
		\item The other one pulls, resolves conflict, then pushes
	\end{itemize}

}
